import os
from IPython.core.display import HTML
import random
map_osm = folium.Map(location=[39.29, -76.61], zoom_start=11)
Holder = []
counter = 0
arrests_lebron = FeatureGroup(name='Sex')
arrests_lechuga = FeatureGroup(name='Race')
length_to_use = len(arrest_table['race'])
while counter < 150:
    curr_num = random.randint(0, length_to_use)
    if curr_num not in Holder:
        counter += 1
        Holder.append(curr_num)
        
num_fem = 0
num_men = 0
for i in Holder:
    if arrest_table['sex'].values[i] == 'M':
        num_men +=1
        arrests_lebron.add_child(folium.Marker(location=[arrest_table['lat'].values[i],arrest_table['long'].values[i]], popup="Arrest #: " + str(arrest_table['arrest'].values[i]) + ". Arrest Location: " + arrest_table['arrestLocation'].values[i] + ". Age: " + str(arrest_table['age'].values[i]), icon=folium.Icon(color='blue')))
    else:
        num_fem +=1
        arrests_lebron.add_child(folium.Marker(location=[arrest_table['lat'].values[i],arrest_table['long'].values[i]],popup="Arrest #: " + str(arrest_table['arrest'].values[i]) + ". Arrest Location: " + arrest_table['arrestLocation'].values[i] + ". Age: " + str(arrest_table['age'].values[i]),icon=folium.Icon(color='red')))
        
print("Male(blue):" + str(num_men))        
print("Female(pink):" + str(num_fem))  

num_black = 0
num_white = 0

for i in Holder:
    if arrest_table['race'].values[i] == 'W':
        num_white +=1
        arrests_lechuga.add_child(folium.Circle(radius=100,location=[arrest_table['lat'].values[i],arrest_table['long'].values[i]],popup="Race : " + str(arrest_table['race'].values[i]), color='white', fill=True,))
    elif arrest_table['race'].values[i] == 'B':
        num_black +=1
        arrests_lechuga.add_child(folium.Circle(radius=100,location=[arrest_table['lat'].values[i],arrest_table['long'].values[i]],popup="Race : " + str(arrest_table['race'].values[i]), color='black', fill=True,))



print("Black(Arrested):" + str(num_black))        
print("White(Arrested):" + str(num_white)) 
map_osm.add_child(arrests_lechuga)
map_osm.add_child(arrests_lebron)


map_osm.save(os.path.join('map_osm.html'))
